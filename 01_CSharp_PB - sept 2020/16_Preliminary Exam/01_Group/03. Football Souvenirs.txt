using System;

namespace FootballSouvenirs
{
    class Program
    {
        static void Main(string[] args)
        {
            string team = Console.ReadLine();
            string souvenir = Console.ReadLine();
            int countSouvenirs = int.Parse(Console.ReadLine());

            double priceSouvenirs = 0;                       

            if (team == "Argentina")
            {
                if (souvenir == "flags")
                {
                    priceSouvenirs = countSouvenirs * 3.25;
                }
                else if (souvenir == "caps")
                {
                    priceSouvenirs = countSouvenirs * 7.20;
                }
                else if (souvenir == "posters")
                {
                    priceSouvenirs = countSouvenirs * 5.10;
                }
                else if (souvenir == "stickers")
                {
                    priceSouvenirs = countSouvenirs * 1.25;
                }
                else 
                {
                    Console.WriteLine("Invalid stock!");
                }
                Console.WriteLine($"Pepi bought {countSouvenirs} {souvenir} of {team} for {priceSouvenirs:F2} lv.");
            }

            else if (team == "Brazil")
            {
                if (souvenir == "flags")
                {
                    priceSouvenirs = countSouvenirs * 4.20;
                }
                else if (souvenir == "caps")
                {
                    priceSouvenirs = countSouvenirs * 8.50;
                }
                else if (souvenir == "posters")
                {
                    priceSouvenirs = countSouvenirs * 5.35;
                }
                else if (souvenir == "stickers")
                {
                    priceSouvenirs = countSouvenirs * 1.20;
                }
                else
                {
                    Console.WriteLine("Invalid stock!");
                }
                Console.WriteLine($"Pepi bought {countSouvenirs} {souvenir} of {team} for {priceSouvenirs:F2} lv.");
            }

            else if (team == "Croatia")
            {
                if (souvenir == "flags")
                {
                    priceSouvenirs = countSouvenirs * 2.75;
                }
                else if (souvenir == "caps")
                {
                    priceSouvenirs = countSouvenirs * 6.90;
                }
                else if (souvenir == "posters")
                {
                    priceSouvenirs = countSouvenirs * 4.95;
                }
                else if (souvenir == "stickers")
                {
                    priceSouvenirs = countSouvenirs * 1.10;
                }
                else
                {
                    Console.WriteLine("Invalid stock!");
                }
                Console.WriteLine($"Pepi bought {countSouvenirs} {souvenir} of {team} for {priceSouvenirs:F2} lv.");
            }

            else if (team == "Denmark")
            {
                if (souvenir == "flags")
                {
                    priceSouvenirs = countSouvenirs * 3.10;
                }
                else if (souvenir == "caps")
                {
                    priceSouvenirs = countSouvenirs * 6.50;
                }
                else if (souvenir == "posters")
                {
                    priceSouvenirs = countSouvenirs * 4.80;
                }
                else if (souvenir == "stickers")
                {
                    priceSouvenirs = countSouvenirs * 0.90;
                }
                else
                {
                    Console.WriteLine("Invalid stock!");
                }
                Console.WriteLine($"Pepi bought {countSouvenirs} {souvenir} of {team} for {priceSouvenirs:F2} lv.");
            }

            else
            {
                Console.WriteLine("Invalid country!");
            }            
        }
    }
}
