using System;

namespace Moon
{
    class Program
    {
        static void Main(string[] args)
        {
            double averageSpeed = double.Parse(Console.ReadLine());
            double litre = double.Parse(Console.ReadLine());

            double distance = 384400 * 2;
            double time = Math.Ceiling(distance / averageSpeed);
            double allTime = time + 3;
            double fuel = (litre * distance)/100;

            Console.WriteLine(allTime);
            Console.WriteLine(fuel);
        }
    }
}
