using System;
using System.ComponentModel;

namespace ComputerRoom
{
    class Program
    {
        static void Main(string[] args)
        {
            string month = Console.ReadLine();
            int hoursSpent = int.Parse(Console.ReadLine());
            int countPeople = int.Parse(Console.ReadLine());
            string dayTime = Console.ReadLine(); // day or night

            double price = 0;
            double totalPriceForPerson = 0;
            double totalPrice = 0;


            if (month == "march" || month == "april" || month == "may")
            {
                if (dayTime == "day")
                {
                    price = hoursSpent * 10.50;
                }
                else if (dayTime == "night")
                {
                    price = hoursSpent * 8.40;
                }
            }
            else if (month == "january" || month == "february" || month == "june" || month == "july" || month == "august"
                 || month == "september" || month == "octomber" || month == "november" || month == "december")
            {
                if (dayTime == "day")
                {
                    price = hoursSpent * 12.60;
                }
                else if (dayTime == "night")
                {
                    price = hoursSpent * 10.20;
                }
            }


            if (countPeople >= 4 && hoursSpent >= 5)
            {
                totalPriceForPerson = price - (price * 0.10) - ((price - (price * 0.10))/2);
                totalPrice = totalPriceForPerson * countPeople;
            }
            else if (countPeople >= 4)
            {
                totalPriceForPerson = price - (price * 0.10);
                totalPrice = totalPriceForPerson * countPeople;
            }
            else if (hoursSpent >= 5)
            {
                totalPriceForPerson = price * 0.50;
                totalPrice = totalPriceForPerson * countPeople;
            }
            else
            {
                totalPriceForPerson = price;
                totalPrice = totalPriceForPerson * countPeople;
            }

            Console.WriteLine($"Price per person for one hour: {totalPriceForPerson / hoursSpent:F2}");
            Console.WriteLine($"Total cost of the visit: {totalPrice:F2}");
        }
    }
}
