using System;

namespace FinalCompetition
{
    class Program
    {
        static void Main(string[] args)
        {
            int countDancers = int.Parse(Console.ReadLine());
            double countPoints = double.Parse(Console.ReadLine());
            string season = Console.ReadLine();
            string place = Console.ReadLine();
            
            double priceSeason = 0;            

            if (place == "Bulgaria")
            {
                double moneyPrize = countPoints * countDancers;
                
                if (season == "summer")
                {
                    priceSeason = moneyPrize * 0.95;
                }
                else
                {
                    priceSeason = moneyPrize * 0.92;
                }
            }

            else if (place == "Abroad")
            {
                double moneyPrize = countPoints * countDancers + (countPoints * countDancers) * 0.50;
                
                if (season == "summer")
                {
                    priceSeason = moneyPrize * 0.90;
                }
                else
                {
                    priceSeason = moneyPrize * 0.85;
                }
            }

            double priceReturn = priceSeason * 0.25;
            double moneyForOneDancer = priceReturn / countDancers;

            Console.WriteLine($"Charity - {priceSeason - priceReturn:F2}");
            Console.WriteLine($"Money per dancer - {moneyForOneDancer:F2}");
        }
    }
}
