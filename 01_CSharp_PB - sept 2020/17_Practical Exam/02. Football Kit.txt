using System;

namespace FootballKit
{
    class Program
    {
        static void Main(string[] args)
        {
            double priceTshirt = double.Parse(Console.ReadLine());
            double totalSum = double.Parse(Console.ReadLine());

            double priceShorts = priceTshirt * 0.75;
            double priceSocks = priceShorts * 0.20;
            double priceButtons = (priceTshirt + priceShorts) * 2.0;

            double sum = (priceTshirt + priceShorts + priceSocks + priceButtons) * 0.85;

            if (sum >= totalSum)
            {
                Console.WriteLine($"Yes, he will earn the world-cup replica ball!");
                Console.WriteLine($"His sum is {sum:F2} lv.");
            }

            else
            {
                Console.WriteLine($"No, he will not earn the world-cup replica ball.");
                Console.WriteLine($"He needs {totalSum - sum:F2} lv. more.");
            }
        }
    }
}
