using System;
using System.Linq;

namespace Zig_Zag_Arrays
{
    class Program
    {
        static void Main(string[] args)
        {
            int size = int.Parse(Console.ReadLine()); // Зададен брой редове, които ще получим

            int[] first = new int[size]; 
            int[] second = new int[size];

            for (int i = 0; i < size; i++)
            {
                int[] numbers = Console.ReadLine()
                    .Split() // Разстояние между числата
                    .Select(int.Parse) // Преобразуваме текста в целочислени числа
                    .ToArray(); // Числата ги добавяме в макив

                int firstNumber = numbers[0]; // Нова променлива с първи индекст
                int secondNumber = numbers[1]; // Нова променлива с втори индекст

                if (i % 2 != 0) // С if проверяваме реда на въвеждане на числата
                {
                    // Ако редът е четен, то първият масив взима второто въведено [1], 
                    // а вторият - първото въведено [0]
                    first[i] = secondNumber;
                    second[i] = firstNumber;
                }
                
                else // Ако не е четен реда, то числата се взимат в обратен ред
                {
                    first[i] = firstNumber;
                    second[i] = secondNumber;
                }
            }

            Console.WriteLine(string.Join(" ", first)); // Взима елементите и прави стринг,
            Console.WriteLine(string.Join(" ", second)); // като след всеки елемент добавя спейс.
        }
    }
}
