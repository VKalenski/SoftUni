using System;
using System.Linq;

namespace Top_Integers
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] numbers = Console.ReadLine()
                .Split()
                .Select(int.Parse)
                .ToArray();

            for (int i = 0; i < numbers.Length; i++) 
                // Преминаваме през всички числа в масива
            {
                int currentNumber = numbers[i];
                bool isTopNumber = true; // Приемаме в началото, че всяко число
                                         // от масива е "ТОП ЧИСЛО".

                for (int j = i + 1; j < numbers.Length; j++)
                    // Започваме от текущия индекс + 1, т.е. първото ни дясно число.
                    // Така ние хващаме всички десни числя спрямо currentNumber.
                {
                    int rightNumber = numbers[j];

                    if (rightNumber >= currentNumber)
                        // Проверяваме дали дясното число е по-голямо от текущото
                        // и ако е по-голямо, то това означава, че нашето число не е "ТОП ЧИСЛО",
                        // и ако не е такова, то ние го сетваме на False и прекъсваме цикъла, 
                        // за да може да провери следващото число.
                    {
                        isTopNumber = false;
                        break;
                    }
                }
                if (isTopNumber) // Ако числата са "ТОП" си ги отпечатваме на конзолата.
                {
                    Console.Write($"{currentNumber} ");
                }
            }

            Console.WriteLine();
        }
    }
}
